# 1) Задание
Заполнить двумернный массив. Все элементы выше главной диагонали равны 0, но все элементы на гл. диагонали и ниже заполняются одинаковыми цифрами (1-9).
# 2) Код программы

```cpp
#include <iostream>
using namespace std;
int main()
{
    const int n = 6;
    int a[n][n];
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++) 
        {
            a[i][j] = 0;
        }
    }


    int m = 1, t;
    for (int i = n; i > -1; i--)
    {
        t = m;
        for (int j = n; j >= i; j--)  
        {
            a[j][i] = t;
            t--;
        }
        m++;
    }
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
```
